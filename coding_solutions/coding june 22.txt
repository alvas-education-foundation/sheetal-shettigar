Program to rotate bits of number in python

INT_BITS = 32
def leftRotate(n, d): 
   return (n << d)|(n >> (INT_BITS - d)) 
def rightRotate(n, d): 
   return (n >> d)|(n << (INT_BITS - d)) & 0xFFFFFFFF
